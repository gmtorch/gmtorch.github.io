def pdm(d: int) -> torch.Tensor:
    """
    Return a symmetric positive definite matrix of n d x d.

    Args:
        d (int): The dimension of the symmetric positive defninite matrix

    Returns
    -------
        _type_: torch.tensor of dimension d x d
    """
    assert d > 0
    shape = (d, d)
    a = torch.tril(
        torch.rand(shape)
    )  # Maybe even more efficient if generating only the needed amounts
    b = a @ a.T
    b = 0.5 * (b + b.T)  # ensures symmetry
    b = b + 1e-3 * torch.eye(d)  # For floating point inprecision in higher dimension
    return b

def random_weights(n: int) -> torch.Tensor:
    """
    Return n weights that add up to 1.

    Args:
        n (int): number of weights

    Returns
    -------
        _type_: torch.tensor of size n
    """
    assert n > 0
    weights = torch.rand(n)
    weights = weights / torch.sum(weights)
    assert torch.sum(weights) == 1
    return weights

class GaussianMixture:
    d: int
    K: int
    pi: torch.Tensor
    means: torch.Tensor
    deviations: torch.Tensor
    threshhold: float = 1e-4

    def __init__(
        self,
        d: int,
        k: int,
        pi: torch.Tensor,
        mean: torch.Tensor | None = None,
        deviations: torch.Tensor | None = None,
    ) -> None:
        """
        Return Gaussian mixture.

        Args:
            d (int): Dimension
            K (int): Number of Gaussian distributions in the Gaussian mixture.
            pi (torch.Tensor): Tensor of weights that should add up to 1.
            mean (torch.Tensor, optional): Mean of the Gaussian mixture.
            deviations (torch.Tensor, optional): Deviations of the Gaussian mixture.
        """
        assert d > 0
        assert k > 0
        assert abs(torch.sum(pi).item() - 1.0) < self.threshhold, (
            f"Expected sum of 1, got {torch.sum(pi).item()}"
        )  # Threshhold not optimized
        assert check_shape(pi, torch.empty(k).shape), (
            f"Expected pi with shape {torch.empty(k).shape}, got {pi.shape}"
        )
        assert mean is None or check_shape(mean, torch.empty(d, k).shape), (
            f"Expected mean with shape {torch.empty(d, k).shape}, got {mean.shape}"
        )
        assert deviations is None or check_shape(deviations, torch.empty(k, d, d).shape), (
            f"Expected mean with shape {torch.empty(k, d, d).shape}, got {deviations.shape}"
        )

        self.d = d
        self.K = k
        self.pi = pi
        self.means = torch.zeros(k, d) if mean is None else mean

        if deviations is None:
            self.deviations = (
                torch.tensor([1]) if d == 1 else torch.stack([torch.eye(d) for _ in range(k)])
            )
        else:
            self.deviations = deviations

    @classmethod
    def random(cls, d: int, k: int, mean_range: int) -> GaussianMixture:
        """
        Return Gaussian mixture with randomly generated mean and derivation.

        Args:
            d (int): Dimension
            K (int): Number of Gaussian distributions in the Gaussian mixture
            mean_range (int): Max value that the mean can take.

        Returns
        -------
            _type_: Gaussian mixture of type gm
        """
        assert mean_range >= 1
        mean = torch.mul(torch.rand(k, d), mean_range)
        pi = random_weights(k)
        deviation = []
        for _ in range(k):
            if d == 1:
                deviation.append(torch.rand(1))
            else:
                deviation.append(pdm(d))
        deviations = torch.stack(deviation, 0)

        return cls(d, k, pi, mean, deviations)

    def sample(self, n: int) -> torch.Tensor:
        """
        Return #n_samples of samples.

        Args:
            n (int): number of samples

        Returns
        -------
            _type_: torch.Tensor filled with n_samples samples of the Gaussian mixture
        """
        assert n > 0
        indices = torch.multinomial(
            self.pi, n, replacement=True
        )  # Chooses from which of the K Gaussians to sample from
        samples = []

        for i in indices:
            if self.d == 1:
                distribution: torch.distributions.Distribution = torch.distributions.normal.Normal(
                    self.means[i.item()], self.deviations[i.item()]
                )
            else:
                distribution = torch.distributions.multivariate_normal.MultivariateNormal(
                    self.means[i.item()], self.deviations[i.item()]
                )
            samples.append(distribution.sample())
        result = torch.stack(samples)
        return result

    def mean(self, n: int) -> torch.Tensor:
        """
        Return the mean of the Gaussian mixture.

        Args:
            n (int): number of samples

        Returns
        -------
            _type_: torch.Tensor
        """
        sample = self.sample(n)
        mean = torch.mean(sample, dim=0)
        return mean

    def avg_mean(self, n: int, loops: int) -> torch.Tensor:
        """
        Return the average of #loops means.

        Args:
            n (int): number of samples
            loops (int): number of means to calculate the average from

        Returns
        -------
            _type_: torch.Tensor
        """
        means = torch.stack([self.mean(n) for _ in range(loops)])
        avg_mean = torch.mean(means, dim=0)
        return avg_mean
